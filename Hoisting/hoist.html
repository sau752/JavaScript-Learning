JavaScript Hoisting

            Hoisting is JavaScript interpreter's action of declaring all variables and functions at the top of the current scope.

Hoist Variable 1


         alert(a);  //is not defined

Hoist Variable 2


        var a;
        alert(a); //Undefined 
        a = 5;

Hoist Variable 3


          alert(a);  //Undefined
          var a = 5;

Hoist Variable 4


        var name = "Saurabh";
        (function () {
            alert(name); //Undefined
            var name = "Milind"
            alert(name); //Milind
        }());

Function Hoisting 1


        hoist();
        function hoist() {
            alert("function is defined later");
        }

Function Hoisting 2


        hoist();    // hoist is not a function
        var hoist = function () {
            alert("function is defined later");
        }

Function Hoisting 3


        abc();  // abc is not defined
        hoist();    // undefined is not a function
        var hoist = function abc() {
            alert("function is defined later");
        }
